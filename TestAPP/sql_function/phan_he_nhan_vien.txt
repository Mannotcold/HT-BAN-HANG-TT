--Xem danh sách hợp đồng của đối tác:
SELECT * FROM HOPDONG;
--Xem thống kê số lượng khách hàng của từng đối tác theo từng ngày, tháng, năm:
SELECT count(DISTINCT MAKH) FROM DONHANG d GROUP BY MADT, NGAYLAP;
SELECT count(DISTINCT MAKH) FROM DONHANG d GROUP BY MADT, DATETRUNC(MONTH, NGAYLAP);
SELECT count(DISTINCT MAKH) FROM DONHANG d GROUP BY MADT, DATETRUNC(YEAR, NGAYLAP);
--Xem danh sách hợp đồng sắp hết hạn
SELECT * FROM HOPDONG WHERE THOIGIANHIEULUC >= DATEADD(MONTH, -1, CURRENT_TIMESTAMP);
--Thống kê số lượng đơn hàng, doanh thu của từng đối tác:
SELECT count(*) FROM DONHANG GROUP BY MADT;
SELECT sum(TONGTIEN) FROM DONHANG GROUP BY MADT;
